☕
Java fue creado por James Gosling en Sun Microsystems en 1995. Su lema es "Write Once, Run Anywhere" (WORA): el código compilado en un sistema funciona en cualquier otro con JVM. Hoy es el lenguaje líder en sistemas empresariales, Android (hasta Kotlin), servidores y big data.

Los 4 pilares de la POO

🏗️

Encapsulamiento

Ocultar el estado interno. Los datos son private, el acceso es por getters/setters públicos. Protege la integridad del objeto.

private / public / protected
🧬

Herencia

Una clase hijo hereda atributos y métodos del padre con extends. Reutiliza código y establece jerarquías. Java solo permite herencia simple.

extends / super
🎭

Polimorfismo

El mismo método se comporta diferente según el objeto. Sobreescritura (@Override) en herencia. Sobrecarga con mismo nombre y distintos parámetros.

@Override / sobrecarga
🔲

Abstracción

Modela conceptos complejos mostrando solo lo esencial. abstract class puede tener métodos abstractos e implementados. interface define solo el contrato.

abstract / interface
Java — Clase completa con encapsulamiento 
public class CuentaBancaria {
    // Atributos privados (encapsulamiento)
    private String titular;
    private double saldo;
    private static int totalCuentas = 0; // pertenece a la clase

    // Constructor
    public CuentaBancaria(String titular, double saldoInicial) {
        this.titular = titular;
        this.saldo = saldoInicial;
        totalCuentas++;
    }

    // Constructor sobrecargado
    public CuentaBancaria(String titular) {
        this(titular, 0.0); // llama al constructor anterior
    }

    // Getters
    public String getTitular() { return titular; }
    public double getSaldo()   { return saldo; }
    public static int getTotalCuentas() { return totalCuentas; }

    // Setter con validación
    public void setTitular(String t) {
        if (t == null || t.isBlank()) throw new IllegalArgumentException("Nombre vacío");
        this.titular = t;
    }

    // Métodos de negocio
    public void depositar(double monto) {
        if (monto <= 0) throw new IllegalArgumentException("Monto inválido");
        saldo += monto;
    }

    public void retirar(double monto) {
        if (monto > saldo) throw new IllegalStateException("Saldo insuficiente");
        saldo -= monto;
    }

    // toString para representación textual
    @Override
    public String toString() {
        return String.format("Cuenta[%s, $%.2f]", titular, saldo);
    }
}

Java — Herencia y Polimorfismo 
// ── CLASE BASE ────────────────────────
public abstract class Figura {
    protected String color;

    public Figura(String color) { this.color = color; }

    // Método abstracto: DEBE implementarse en subclases
    public abstract double area();
    public abstract double perimetro();

    // Método concreto: heredado tal cual
    public void describir() {
        System.out.printf("%s %s: área=%.2f, perímetro=%.2f%n",
                           color, getClass().getSimpleName(), area(), perimetro());
    }
}

// ── SUBCLASES ─────────────────────────
public class Circulo extends Figura {
    private double radio;
    public Circulo(String color, double r) { super(color); radio = r; }
    @Override public double area()      { return Math.PI * radio * radio; }
    @Override public double perimetro() { return 2 * Math.PI * radio; }
}

public class Rectangulo extends Figura {
    private double ancho, alto;
    public Rectangulo(String c, double a, double b) { super(c); ancho=a; alto=b; }
    @Override public double area()      { return ancho * alto; }
    @Override public double perimetro() { return 2 * (ancho + alto); }
}

// ── POLIMORFISMO ──────────────────────
Figura[] figuras = {
    new Circulo("Rojo", 5),
    new Rectangulo("Azul", 4, 6),
    new Circulo("Verde", 3)
};
for (Figura f : figuras) f.describir(); // cada una imprime su propio cálculo

// instanceof para verificar tipo
if (figuras[0] instanceof Circulo circulo) {
    System.out.println("Radio: " + circulo.radio); // Pattern matching Java 16+
}
Java — Interfaces y herencia múltiple 
// ── INTERFACES ────────────────────────
interface Serializable {
    String serializar(); // abstracto por defecto
}

interface Validable {
    boolean esValido();
    // Método default: tiene implementación por defecto
    default void validarOLanzar() {
        if (!esValido()) throw new IllegalStateException("Objeto inválido");
    }
}

// Una clase puede implementar MÚLTIPLES interfaces
public class Usuario implements Serializable, Validable {
    private String email;

    @Override public String serializar() {
        return "{"email":"" + email + ""}";
    }

    @Override public boolean esValido() {
        return email != null && email.contains("@");
    }
}

Regla de oro: Prefiere composición sobre herencia cuando sea posible. La herencia crea acoplamiento fuerte; la composición ("tiene un") es más flexible que "es un".

// Una interfaz funcional tiene exactamente UN método abstracto

import java.util.stream.*;

List<String> lenguajes = List.of(
    "Java", "Python", "JavaScript", "TypeScript",
    "Kotlin", "Go", "Rust", "C++", "Java"
);

// ── OPERACIONES INTERMEDIAS (lazy) ────
lenguajes.stream()
    .filter(l -> l.length() > 3)          // solo longitud > 3
    .map(String::toUpperCase)               // transformar
    .distinct()                             // eliminar duplicados

Java — Proyecto completo: Sistema de Biblioteca 
import java.util.*;
import java.util.stream.*;

// Modelo
record Libro(String titulo, String autor, int anio, String genero, double precio) {
    // Java 16+: record genera constructor, getters, equals, hashCode, toString
}

public class Biblioteca {
    private final List<Libro> catalogo = new ArrayList<>();

    public void agregar(Libro l) { catalogo.add(l); }

    // Buscar por autor (ignora mayúsculas)
    public List<Libro> porAutor(String autor) {
        return catalogo.stream()
            .filter(l -> l.autor().equalsIgnoreCase(autor))
            .collect(Collectors.toList());
    }

    // Top N libros más baratos de un género
    public List<Libro> topBaratos(String genero, int n) {
        return catalogo.stream()
            .filter(l -> l.genero().equals(genero))
            .sorted(Comparator.comparingDouble(Libro::precio))
            .limit(n)
            .collect(Collectors.toList());
    }

    // Estadísticas por género
    public Map<String, DoubleSummaryStatistics> estadisticasPorGenero() {
        return catalogo.stream()
            .collect(Collectors.groupingBy(Libro::genero,
                Collectors.summarizingDouble(Libro::precio)));
    }

    // Reportar todos los géneros disponibles (sin duplicados, ordenados)
    public List<String> generos() {
        return catalogo.stream()
            .map(Libro::genero)
            .distinct()
            .sorted()
            .collect(Collectors.toList());
    }
}

¿Qué es HTML y cómo funciona?

🌐
HTML (HyperText Markup Language) es el lenguaje de marcado que estructura el contenido web. No es un lenguaje de programación — es una colección de etiquetas que el navegador interpreta para mostrar texto, imágenes, formularios, etc. Fue creado por Tim Berners-Lee en 1991. Hoy usamos HTML5 (2014).

🎨
CSS (Cascading Style Sheets) controla la apariencia visual de HTML. "Cascading" significa que los estilos se aplican en cascada: el navegador, el autor y el usuario. La especificidad y el orden determinan qué estilo gana.
CSS — Box Model 
/* El modelo de caja: todo elemento es una caja */
.caja {
    width: 300px;         /* ancho del contenido */
    height: 200px;        /* alto del contenido */
    padding: 20px;        /* espacio interior (empuja hacia afuera) */
    border: 2px solid red;
    margin: 10px auto;    /* auto centra horizontalmente */

    /* Sin esto: width = contenido+padding+border = 344px */
    /* Con esto: width incluye todo = siempre 300px */
    box-sizing: border-box;
}

/* Aplicar border-box a todo (recomendado) */
*, *::before, *::after { box-sizing: border-box; }

/* Padding individual */
padding-top: 10px;
padding-right: 20px;
padding-bottom: 10px;
padding-left: 20px;
/* Shorthand: top right bottom left (sentido horario) */
padding: 10px 20px 10px 20px;
padding: 10px 20px;  /* top/bottom | left/right */
padding: 10px;       /* todos lados */

Flexbox (CSS3, 2012) es un modelo de layout unidimensional: trabaja en una sola dirección (fila o columna). Ideal para distribuir elementos dentro de un contenedor, alinearlos y gestionar el espacio entre ellos. Soportado en todos los navegadores modernos.
CSS — Flexbox: contenedor completo 
/* ── ACTIVAR FLEXBOX ────────────────── */
.contenedor {
    display: flex;          /* o inline-flex */

    /* DIRECCIÓN DEL EJE PRINCIPAL */
    flex-direction: row;           /* → derecha (default) */
    flex-direction: row-reverse;   /* ← izquierda */
    flex-direction: column;        /* ↓ abajo */
    flex-direction: column-reverse;/* ↑ arriba */

    /* SALTO DE LÍNEA */
    flex-wrap: nowrap;   /* sin salto (default) */
    flex-wrap: wrap;     /* salta a la siguiente línea */
    flex-wrap: wrap-reverse;

    /* Shorthand: flex-direction + flex-wrap */
    flex-flow: row wrap;

    /* ALINEACIÓN EN EJE PRINCIPAL (justify-content) */
    justify-content: flex-start;    /* |||| ... (default) */
    justify-content: flex-end;      /* ... |||| */
    justify-content: center;        /* .. |||| .. */
    justify-content: space-between; /* | ... | ... | */
    justify-content: space-around;  /* . | .. | .. | . */
    justify-content: space-evenly;  /* . | . | . | . */

    /* ALINEACIÓN EN EJE SECUNDARIO (align-items) */
    align-items: stretch;      /* ocupa todo el alto (default) */
    align-items: flex-start;   /* arriba */
    align-items: flex-end;     /* abajo */
    align-items: center;       /* centrado vertical */
    align-items: baseline;     /* alinea por texto base */

    /* ALINEACIÓN DE LÍNEAS (cuando hay wrap) */
    align-content: flex-start;
    align-content: center;
    align-content: space-between;
    align-content: stretch;     /* (default) */

    /* ESPACIO ENTRE ELEMENTOS */
    gap: 1rem;             /* fila y columna */
    gap: 1rem 2rem;       /* fila | columna */
    row-gap: 1rem;
    column-gap: 2rem;
}
CSS — Flexbox: propiedades de los hijos 
.hijo {
    /* ORDEN VISUAL (sin cambiar el HTML) */
    order: 0;       /* default: 0. Menor = aparece antes */
    order: -1;      /* aparece el primero */
    order: 99;      /* aparece el último */

    /* CRECIMIENTO (ocupa espacio libre) */
    flex-grow: 0;   /* no crece (default) */
    flex-grow: 1;   /* crece para llenar espacio */
    /* Si hijo-A tiene flex-grow:1 y hijo-B flex-grow:2,
       B recibirá el doble de espacio libre que A */

    /* ENCOGIMIENTO (reduce si no hay espacio) */
    flex-shrink: 1;  /* se encoge (default) */
    flex-shrink: 0;  /* no se encoge nunca */

    /* TAMAÑO BASE */
    flex-basis: auto;    /* usa width/height (default) */
    flex-basis: 200px;
    flex-basis: 30%;

    /* SHORTHAND: flex-grow flex-shrink flex-basis */
    flex: 1;           /* flex: 1 1 0% */
    flex: auto;        /* flex: 1 1 auto */
    flex: none;        /* flex: 0 0 auto */
    flex: 0 0 200px;  /* fijo en 200px */

    /* SOBREESCRIBIR align-items para este hijo */
    align-self: auto;
    align-self: flex-start;
    align-self: flex-end;
    align-self: center;
    align-self: stretch;
}

CSS Grid (2017) es un modelo de layout bidimensional: maneja filas Y columnas al mismo tiempo. Es ideal para layouts de páginas completas. Complementa a Flexbox: usa Grid para el macro-layout, Flexbox para el micro-layout.
CSS — Grid: contenedor completo 
.grid {
    display: grid;  /* o inline-grid */

    /* DEFINIR COLUMNAS */
    grid-template-columns: 200px 1fr 1fr;        /* 3 col: fija + 2 iguales */
    grid-template-columns: repeat(3, 1fr);       /* 3 columnas iguales */
    grid-template-columns: repeat(auto-fill, minmax(250px, 1fr)); /* responsivo */
    grid-template-columns: repeat(auto-fit,  minmax(200px, 1fr));  /* se estira */
    grid-template-columns: 1fr 2fr 1fr;          /* proporciones */

    /* DEFINIR FILAS */
    grid-template-rows: 80px 1fr auto;    /* header + main + footer */
    grid-auto-rows: minmax(100px, auto); /* filas implícitas */

    /* ÁREAS CON NOMBRE */
    grid-template-areas:
        "header  header  header"
        "sidebar main    main  "
        "footer  footer  footer";

    /* ESPACIO ENTRE CELDAS */
    gap: 1rem;           /* row-gap + column-gap */
    row-gap: 1rem;
    column-gap: 2rem;

    /* ALINEACIÓN DE TODO EL GRID */
    justify-content: center;    /* grid dentro del contenedor */
    align-content: center;

    /* ALINEACIÓN DE ITEMS DENTRO DE CELDAS */
    justify-items: stretch;   /* horizontal (default) */
    align-items: stretch;     /* vertical (default) */
    place-items: center;      /* shorthand: align + justify */
}

abstract class Servicio<T> {
    protected abstract validar(datos: Partial<T>): boolean;
    abstract crear(datos: Omit<T, 'id'>): Promise<T>;

    async crearConValidacion(datos: Omit<T, 'id'>): Promise<T> {
        if (!this.validar(datos)) {
            throw new Error("Datos inválidos");
        }
        return this.crear(datos);
    }
}

interface Producto { id: number; nombre: string; precio: number; }

class ProductoServicio extends Servicio<Producto> {
    private db: Producto[] = [];

    protected validar(datos: Partial<Producto>): boolean {
        return !!datos.nombre && (datos.precio ?? -1) > 0;
    }

    async crear(datos: Omit<Producto, 'id'>): Promise<Producto> {
        const nuevo = { id: Date.now(), ...datos };
        this.db.push(nuevo);
        return nuevo;
    }

    buscar(query: string): Producto[] {
        return this.db.filter(p =>
            p.nombre.toLowerCase().includes(query.toLowerCase())
        );
    }
}

JOINs: combinar tablas

🔗
Los JOINs relacionan datos de múltiples tablas. INNER JOIN solo devuelve coincidencias. LEFT JOIN devuelve todos de la izquierda + coincidencias. RIGHT JOIN es lo opuesto. FULL OUTER JOIN no existe en MySQL (usa UNION).
MySQL — INNER, LEFT, RIGHT JOIN 
-- Tablas de ejemplo:
-- usuarios: id, nombre, email
-- productos: id, nombre, precio, usuario_id, categoria_id
-- categorias: id, nombre
-- pedidos: id, usuario_id, total, fecha
-- pedido_items: id, pedido_id, producto_id, cantidad, precio_unit

-- ── INNER JOIN (solo coincidencias) ──────────
SELECT u.nombre, u.email, p.nombre AS producto, p.precio
FROM usuarios u
INNER JOIN productos p ON u.id = p.usuario_id
WHERE p.precio > 100
ORDER BY p.precio DESC;

-- ── LEFT JOIN (todos de la izquierda) ────────
-- Usuarios con o sin productos
SELECT u.nombre,
    COUNT(p.id) AS total_productos,
    COALESCE(SUM(p.precio), 0) AS valor_total
FROM usuarios u
LEFT JOIN productos p ON u.id = p.usuario_id
GROUP BY u.id, u.nombre
ORDER BY total_productos DESC;

-- Encontrar usuarios SIN productos
SELECT u.nombre, u.email
FROM usuarios u
LEFT JOIN productos p ON u.id = p.usuario_id
WHERE p.id IS NULL;

-- ── RIGHT JOIN ────────────────────────────────
-- Productos con o sin categoría (raro: LEFT es más común)
SELECT c.nombre AS categoria, COUNT(p.id) AS productos
FROM productos p
RIGHT JOIN categorias c ON p.categoria_id = c.id
GROUP BY c.id, c.nombre;

-- ── FULL OUTER JOIN (MySQL: usar UNION) ───────
SELECT u.nombre, p.nombre AS producto
FROM usuarios u LEFT JOIN productos p ON u.id = p.usuario_id
UNION
SELECT u.nombre, p.nombre AS producto
FROM usuarios u RIGHT JOIN productos p ON u.id = p.usuario_id
WHERE u.id IS NULL;

-- ── SELF JOIN (tabla consigo misma) ──────────
-- Empleados con su jefe (tabla empleados con columna jefe_id)
SELECT e.nombre AS empleado, j.nombre AS jefe
FROM empleados e
LEFT JOIN empleados j ON e.jefe_id = j.id;

-- ── CROSS JOIN (producto cartesiano) ─────────
-- Todos los colores con todos los tallas
SELECT color, talla
FROM colores
CROSS JOIN tallas;
MySQL — JOINs múltiples y consultas avanzadas 
-- ── JOIN DE 3+ TABLAS ───────────────────────────
SELECT
    u.nombre AS cliente,
    p.id AS pedido_id,
    p.fecha,
    pr.nombre AS producto,
    pi.cantidad,
    pi.precio_unit,
    (pi.cantidad * pi.precio_unit) AS subtotal
FROM pedidos p
INNER JOIN usuarios u         ON p.usuario_id  = u.id
INNER JOIN pedido_items pi    ON p.id           = pi.pedido_id
INNER JOIN productos pr       ON pi.producto_id = pr.id
INNER JOIN categorias c       ON pr.categoria_id = c.id
WHERE p.fecha >= '2024-01-01'
    AND c.nombre = 'Electrónica'
ORDER BY p.fecha DESC, subtotal DESC;

-- ── WINDOW FUNCTIONS (MySQL 8.0+) ────────────
SELECT
    nombre,
    precio,
    categoria_id,
    RANK() OVER (PARTITION BY categoria_id ORDER BY precio DESC) AS rank_precio,
    ROW_NUMBER() OVER (ORDER BY precio DESC) AS fila,
    DENSE_RANK() OVER (PARTITION BY categoria_id ORDER BY precio DESC) AS dense_rank,
    LAG(precio, 1, 0) OVER (PARTITION BY categoria_id ORDER BY precio) AS precio_anterior,
    LEAD(precio) OVER (PARTITION BY categoria_id ORDER BY precio) AS precio_siguiente,
    SUM(precio) OVER (PARTITION BY categoria_id) AS total_categoria,
    AVG(precio) OVER (PARTITION BY categoria_id ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS media_movil
FROM productos;

-- Obtener el producto más caro de cada categoría
SELECT * FROM (
    SELECT *,
        RANK() OVER (PARTITION BY categoria_id ORDER BY precio DESC) AS rn
    FROM productos
) ranked
WHERE rn = 1;

CREATE OR REPLACE VIEW vista_productos_activos AS
    SELECT p.id, p.nombre, p.precio, c.nombre AS categoria, u.nombre AS vendedor
    FROM productos p
    JOIN categorias c ON p.categoria_id = c.id
    JOIN usuarios u   ON p.usuario_id   = u.id
    WHERE p.stock > 0
    ORDER BY p.precio;
